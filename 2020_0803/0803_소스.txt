#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> triangle) {
    int answer = 0;
    vector<vector<int>> ans(triangle.size(),vector<int>(triangle.size()));
    ans[0][0] = triangle[0][0];
    for(int i=1;i<triangle.size();i++){
        for(int j=0;j<=i;j++){
            if(j==0){
                ans[i][j] =ans[i-1][j] + triangle[i][j];
            }else if(i==j){
                ans[i][j] = ans[i-1][j-1] + triangle[i][j];
            }else{
                ans[i][j] = max(ans[i-1][j-1] + triangle[i][j],ans[i-1][j] + triangle[i][j]);
            }
            if(i == triangle.size() - 1){
                answer = max(answer,ans[i][j]);
            }
        }
    }
    return answer;
}
