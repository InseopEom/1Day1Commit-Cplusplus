#include <string>
#include <vector>

using namespace std;

int convert[3] = {1,2,4};

string solution(int n) {



    string str = "";

    while(true){

        n -= 1;

        int tmp = n % 3;

        str = to_string(convert[tmp]) + str;

        n /= 3;

        if(n <= 0){
            break;
        }
    }

    return str;
}
